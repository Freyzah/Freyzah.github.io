enum AdsBannerItemLinkType {
  none,
  product(label: 'Produit'),
  rubric(label: 'Rubrique');

  final String label;
  const AdsBannerItemLinkType({this.label = ''});
}

class AdsBannerItem {
  String imageFileName;
  AdsBannerItemLinkType linkType;
  int? linkId;

  factory AdsBannerItem.fromJson(Map<String, dynamic> json) {
    return AdsBannerItem(
      imageFileName: json['imageFileName'],
      linkType: AdsBannerItemLinkType.values
          .firstWhere((e) => e.name == json['linkType']),
      linkId: json['linkId'],
    );
  }
}
ads_banner_widget.dart — Navigation conditionnelle au clic 
// Quand le client touche une bannière :
if (linkedItem.linkType == AdsBannerItemLinkType.product) {
  // Lien vers un produit → on le cherche par code
  Product? product = productsModel.firstWhere(
    (p) => int.parse(p.code) == linkedItem.linkId);
  productsToShow.add(product!);

} else if (linkedItem.linkType == AdsBannerItemLinkType.rubric) {
  // Lien vers une rubrique → on affiche tous ses produits
  Family? family = familiesModel.firstWhere(
    (f) => int.parse(f.code) == linkedItem.linkId);
  productsToShow.addAll(productsModel.entries
    .where((p) => p.familyCode == family?.code));
}

backup_document.dart — Modèle de sauvegarde 
class BackupDocument {
  static const String baseName = 'backup';

  final int id;
  final String name;
  final String path;
  final DateTime date;

  factory BackupDocument.fromJson(Map<String, dynamic> json) {
    DateTime? docDate = DateTime.tryParse(json['date'] ?? '');
    if (docDate == null) throw Exception('Invalid date');
    return BackupDocument(
      id: json['id'], name: json['name'],
      path: json['path'], date: docDate,
    );
  }

  String get prefix => name.substring(0, name.lastIndexOf('_'));
  static String buildPrefix(int id) => '${baseName}_$id';
}

Future<File?> createBackup(List<File> files) async {
  List<File> toBackup = files.where((f) => f.existsSync()).toList();
  if (toBackup.isEmpty) return null;

  String ts = DateTime.now().toIso8601String()
      .replaceAll(':', '-');
  var encoder = ZipFileEncoder();
  encoder.create('kiosk_tune_backup_$ts.zip');
  for (var file in toBackup) encoder.addFileSync(file);
  encoder.close();
  return File(backupFilePath);
}

Future<bool> restoreBackup(File backupZip) async {
  final bytes = await backupZip.readAsBytes();
  final archive = ZipDecoder().decodeBytes(bytes);
  for (final file in archive) {
    if (!file.isFile) continue;
    File target = File('${localStorageDir.path}/${file.name}');
    await target.parent.create(recursive: true);
    await target.writeAsBytes(file.content);
  }
  return true;
}
